```sql
WITH SHIFT_INFO AS (
	SELECT 
		SH_NM 
		, DT_SH_ENTRY
		, DT_SH_DOCK
		, DT_LT_START
		, CONCAT(SH_CD, '/', SH_YEAR, '/', SH_CHASU) AS SHIP_KEY
		
	FROM 
		tb_dock_shift
	WHERE SH_CD = 'ATE'
		AND MW_CD = 'MW01'
	
), DOCK_WORK_INFO AS (
	SELECT 
		SH_NM
		,DT_PORT_ENTRY
		, DT_DOCK
		, DT_UNLOAD
		, CONCAT(SH_CD, '/', SH_YEAR,'/', SH_CHASU) AS SHIP_KEY       
	FROM tb_dock_work
	WHERE DW_KEY = '20240625-012'
		AND SH_CD = 'ATE'
		AND MW_CD = 'MW01'
)
SELECT 
	s.*
	, dw.*
FROM SHIFT_INFO s
LEFT JOIN DOCK_WORK_INFO dw
ON s.SHIP_KEY = dw.SHIP_KEY;
```
