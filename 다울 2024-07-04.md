private void HandleUnloadPlanOnClick(Cell cell)

                    {

                }

                return;
            }
            else if (cell.BackColor == BerthColorManager.UnloadPlanBackColorByAi ||
                cell.BackColor == BerthColorManager.UnloadPlanBackColorByWorker)

        }
