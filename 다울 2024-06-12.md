public string SQLWhereIn(ArrayList arr)
{
    string inquery = String.Empty;

    {
        if (String.IsNullOrEmpty(inquery) == false) inquery += ",";
        inquery += string.Format("'{0}'", str);
    }

    return inquery;
}
